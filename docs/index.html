Python

from logballoon import LogBalloon

lb = LogBalloon(
    app_name="FFT Analyzer",
    version="1.0.0",
    endpoint="https://ops.example.com",
)
lb.start()
lb.event(
    "export_complete",
    {"rows": 120, "format": "csv", "site": "factory-a"},
)

Event payload

# Fixed envelope:
#   app, version, installation_id,
#   event, timestamp
#
# Free-form for you:
lb.event("job_done", {
    "duration_ms": 842,
    "operator": "A12",
    "batch_id": "2026-07-22-03",
